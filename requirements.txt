import 'package:flutter/material.dart';

void main() => runApp(MaterialApp(home: MyApp()));

class MyApp extends StatefulWidget {
  @override
  _MyAppState createState() => _MyAppState();
}

class _MyAppState extends State<MyApp> {
  TextEditingController id = TextEditingController();
  String dob = "";
  bool flutter = false, java = false, dotnet = false;
  String result = "";

  pickDate() async {
    DateTime? d = await showDatePicker(
      context: context,
      initialDate: DateTime.now(),
      firstDate: DateTime(2000),
      lastDate: DateTime(2100),
    );
    if (d != null) setState(() => dob = "${d.day}/${d.month}/${d.year}");
  }

  showData() {
    String skills = "";
    if (flutter) skills += "Flutter ";
    if (java) skills += "Java ";
    if (dotnet) skills += ".NET";

    setState(() {
      result = "ID: ${id.text}\nDOB: $dob\nSkills: $skills";
    });
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: Text("Q1")),
      body: Column(children: [
        TextField(controller: id, decoration: InputDecoration(labelText: "ID")),
        IconButton(icon: Icon(Icons.date_range), onPressed: pickDate),
        CheckboxListTile(title: Text("Flutter"), value: flutter, onChanged: (v)=>setState(()=>flutter=v!)),
        CheckboxListTile(title: Text("Java"), value: java, onChanged: (v)=>setState(()=>java=v!)),
        CheckboxListTile(title: Text(".NET"), value: dotnet, onChanged: (v)=>setState(()=>dotnet=v!)),
        ElevatedButton(onPressed: showData, child: Text("Show")),
        Text(result)
      ]),
    );
  }
}

## Q3 – Login with 4-Digit PIN (Multi-Screen)

import 'package:flutter/material.dart';

void main() => runApp(MaterialApp(home: MyApp()));

class MyApp extends StatefulWidget {
  @override
  _MyAppState createState() => _MyAppState();
}

class _MyAppState extends State<MyApp> {
  TextEditingController user = TextEditingController();
  TextEditingController p1 = TextEditingController();
  TextEditingController p2 = TextEditingController();
  TextEditingController p3 = TextEditingController();
  TextEditingController p4 = TextEditingController();
  String result = "";

  login() {
    String pin = p1.text + p2.text + p3.text + p4.text;

    setState(() {
      if (user.text == "admin" && pin == "2121") {
        result = "Welcome";
      } else {
        result = "Error";
      }
    });
  }

  reset() {
    user.clear(); p1.clear(); p2.clear(); p3.clear(); p4.clear();
    setState(()=>result="");
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: Text("Q3")),
      body: Column(children: [
        TextField(controller: user, decoration: InputDecoration(labelText: "Username")),
        Row(children: [
          Expanded(child: TextField(controller: p1)),
          Expanded(child: TextField(controller: p2)),
          Expanded(child: TextField(controller: p3)),
          Expanded(child: TextField(controller: p4)),
        ]),
        ElevatedButton(onPressed: login, child: Text("Login")),
        ElevatedButton(onPressed: reset, child: Text("Reset")),
        Text(result)
      ]),
    );
  }
}


## Q4 – Range Numbers in ListView

import 'package:flutter/material.dart';

void main() => runApp(MaterialApp(home: MyApp()));

class MyApp extends StatefulWidget {
  @override
  _MyAppState createState() => _MyAppState();
}

class _MyAppState extends State<MyApp> {
  TextEditingController start = TextEditingController();
  TextEditingController end = TextEditingController();
  List<int> list = [];

  showRange() {
    list.clear();
    int s = int.parse(start.text);
    int e = int.parse(end.text);

    for (int i = s; i <= e; i++) {
      list.add(i);
    }
    setState(() {});
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: Text("Q4")),
      body: Column(children: [
        TextField(controller: start, decoration: InputDecoration(labelText: "Start")),
        TextField(controller: end, decoration: InputDecoration(labelText: "End")),
        ElevatedButton(onPressed: showRange, child: Text("Show Range")),
        Expanded(child: ListView.builder(
          itemCount: list.length,
          itemBuilder: (c,i)=>Text("${list[i]}")
        ))
      ]),
    );
  }
}

## Q5 – Attendance App

import 'package:flutter/material.dart';

void main() => runApp(MaterialApp(home: MyApp()));

class MyApp extends StatefulWidget {
  @override
  _MyAppState createState() => _MyAppState();
}

class _MyAppState extends State<MyApp> {
  List<String> students = ["Vikas","Rohan","Tanisha","Rucha","Pavan"];
  List<bool> present = [false,false,false,false,false];
  String result = "";

  submit() {
    String p="", a="";
    for(int i=0;i<students.length;i++){
      if(present[i]) p += students[i]+"\n";
      else a += students[i]+"\n";
    }
    setState(()=> result = "Present:\n$p\nAbsent:\n$a");
  }

  reset(){
    present = [false,false,false,false,false];
    setState(()=>result="");
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: Text("Q5")),
      body: Column(children: [
        Expanded(
          child: ListView.builder(
            itemCount: students.length,
            itemBuilder: (c,i)=>CheckboxListTile(
              title: Text(students[i]),
              value: present[i],
              onChanged: (v)=>setState(()=>present[i]=v!)
            ),
          ),
        ),
        ElevatedButton(onPressed: submit, child: Text("Submit")),
        ElevatedButton(onPressed: reset, child: Text("Reset")),
        Text(result)
      ]),
    );
  }
}

  reset(){
    bookings.clear();
    setState(()=>msg="");
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: Text("Q6")),
      body: Column(children: [
        DropdownButton(
          value: selected,
          items: slots.map((e)=>DropdownMenuItem(value:e, child:Text(e))).toList(),
          onChanged: (v)=>setState(()=>selected=v.toString())
        ),
        ElevatedButton(onPressed: book, child: Text("Book")),
        ElevatedButton(onPressed: reset, child: Text("Reset")),
        Text(msg),
        Expanded(child: ListView(children: bookings.map((e)=>Text(e)).toList()))

    return Scaffold(
      appBar: AppBar(title: Text("Q13")),
      body: Column(children: [

        Slider(value: n, min:1, max:3, onChanged:(v){n=v;calc();}),
