  @override
  Widget build(BuildContext context) {

      onPressed: () {
        updateRange();
      },
      child: Text("Show Range"),
    );

    ListView list = ListView(

      children: numbers.map((e)=>Text(e.toString())).toList(),

    MaterialApp materialApp=MaterialApp(home:scaffold,debugShowCheckedModeBanner:false);

    return materialApp;
  }

  void updateRange(){
    numbers.clear();
    int s=int.parse(start.text);
    int e=int.parse(end.text);

    for(int i=s;i<=e;i++){
      numbers.add(i);
    }
    setState(() {});
  }

import 'package:flutter/material.dart';

void main() {
  runApp(const MyApp());
}

class MyApp extends StatefulWidget {
  const MyApp({super.key});

  @override
  State<MyApp> createState() => _MyAppState();
}

class _MyAppState extends State<MyApp> {

  String subject="Maths";

  List<String> maths=["Vikas","Rohan","Tanisha","Rucha","Pavan"];
  List<String> science=["Harsh","Ved","Vihan","Chinar","Dev"];

  List<bool> present=[false,false,false,false,false];

  String presentText="";
  String absentText="";

  int pCount=0;
  int aCount=0;

  @override
  Widget build(BuildContext context) {

    DropdownButton d = DropdownButton(
      value: subject,
      items: ["Maths","Science"].map((e)=>DropdownMenuItem(value:e,child:Text(e))).toList(),
      onChanged: (val){
        setState(() {
          subject=val.toString();
          present=[false,false,false,false,false];
        });
      });

    List<String> students = subject=="Maths"?maths:science;

    ListView list = ListView.builder(
      shrinkWrap: true,
      itemCount: students.length,
      itemBuilder: (context,index){
        return CheckboxListTile(
          title: Text(students[index]),
          value: present[index],
          onChanged: (val){
            setState(() {
              present[index]=val!;
            });
          });
      });

    ElevatedButton submit = ElevatedButton(
      onPressed: () {
        updateAttendance(students);
      },
      child: Text("Submit"),
    );

    ElevatedButton reset = ElevatedButton(
      onPressed: () {
        setState(() {
          presentText="";
          absentText="";
          pCount=0;
          aCount=0;
        });
      },
      child: Text("Reset"),
    );

    Text t1=Text("Present ($pCount):\n$presentText");
    Text t2=Text("Absent ($aCount):\n$absentText");

    Row row=Row(children:[Expanded(child:t1),Expanded(child:t2)]);

    Column column=Column(children:[d,list,submit,reset,row]);

    Container container=Container(padding:EdgeInsets.all(10),child:column);

    AppBar appBar=AppBar(title:Text("Attendance"),backgroundColor:Colors.lightBlue);

    Scaffold scaffold=Scaffold(appBar:appBar,body:container);

    MaterialApp materialApp=MaterialApp(home:scaffold,debugShowCheckedModeBanner:false);

    return materialApp;
  }

  void updateAttendance(List<String> students){
    List<String> p=[];
    List<String> a=[];

    for(int i=0;i<students.length;i++){
      if(present[i]){
        p.add(students[i]);
      }else{
        a.add(students[i]);
      }
    }

    setState(() {
      presentText=p.join("\n");
      absentText=a.join("\n");
      pCount=p.length;
      aCount=a.length;
    });
  }
}
