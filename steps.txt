View.py

from django.shortcuts import render, redirect
from .models import Person
from .forms import PersonForm


def home(request):
    form = PersonForm()
    data = Person.objects.all()

    if request.method == "POST":
        form = PersonForm(request.POST)
        if form.is_valid():
            obj = form.save(commit=False)
            obj.hobbies = ",".join(form.cleaned_data["hobbies"])
            form.save()
            return redirect("/")
    return render(request, "home.html", {"form": form, "data": data})


def delete(request, id):
    person = Person.objects.get(id=id)
    person.delete()
    return redirect("/")


def update(request, id):
    person = Person.objects.get(id=id)
    form = PersonForm(instance=person)

    if request.method == "POST":
        form = PersonForm(request.POST, instance=person)
        if form.is_valid():
            obj = form.save(commit=False)
            obj.hobbies = ",".join(form.cleaned_data["hobbies"])
            form.save()
            return redirect("/")
    return render(request, "update.html", {"form": form})

from django.shortcuts import render, redirect
from django.views import View
from .models import Class
from .forms import ClassForm


class HomeView(View):
    def get(self, request):
        form = ClassForm()
        data = Class.objects.all()
        return render(request, "home.html", {"form": form, "data": data})

    def post(self, request):
        form = ClassForm(request.POST)
        if form.is_valid():
            obj = form.save(commit=False)
            obj.hobbies = ",".join(form.cleaned_data["hobbies"])
            obj.save()
            return redirect("/")
        data = Class.objects.all()
        return render(request, "home.html", {"form": form, "data": data})


class UpdateView(View):
    def get(self, request, id):
        person = Class.objects.get(id=id)
        form = ClassForm(instance=person)
        return render(request, "update.html", {"form": form})

    def post(self, request, id):
        person = Class.objects.get(id=id)
        form = ClassForm(request.POST, instance=person)
        if form.is_valid():
            obj = form.save(commit=False)
            obj.hobbies = ",".join(form.cleaned_data["hobbies"])
            obj.save()
            return redirect("/")
        return render(request, "update.html", {"form": form})


class DeleteView(View):
    def get(self, request, id):
        person = Class.objects.get(id=id)
        person.delete()
        return redirect("/")

from django.db import models

class Author(models.Model):
    name = models.CharField(max_length=100)
    biography = models.TextField()
    birthdate = models.DateField()

    def __str__(self):
        return self.name


class Book(models.Model):
    title = models.CharField(max_length=100)
    isbn = models.CharField(max_length=20)
    published_date = models.DateField()
    summary = models.TextField()
    author = models.ForeignKey(Author, on_delete=models.CASCADE)

    def __str__(self):
        return self.title

from django.shortcuts import render, redirect
from .models import Author, Book

from rest_framework import generics
from .serializers import BookSerializer

# ================= DJANGO VIEWS =================

def add_author(request):
    if request.method == "POST":
        name = request.POST["name"]
        bio = request.POST["bio"]
        birth = request.POST["birth"]

        Author.objects.create(
            name=name,
            biography=bio,
            birthdate=birth
        )
        return redirect("author_list")

    return render(request, "add_author.html")


def author_list(request):
    authors = Author.objects.all()
    return render(request, "author_list.html", {"authors": authors})


def author_detail(request, id):
    author = Author.objects.get(id=id)
    books = Book.objects.filter(author=author)

    return render(request, "author_detail.html", {
        "author": author,
        "books": books
    })


# ================= DRF API =================

class BookListCreate(generics.ListCreateAPIView):
    queryset = Book.objects.all()
    serializer_class = BookSerializer


class BookDetail(generics.RetrieveUpdateAPIView):
    queryset = Book.objects.all()
    serializer_class = BookSerializer

from django.db import models

class Event(models.Model):
    title = models.CharField(max_length=100)
    description = models.TextField()
    date = models.DateField()
    location = models.CharField(max_length=100)
    capacity = models.IntegerField()

    def __str__(self):
        return self.title


class Attendee(models.Model):
    event = models.ForeignKey(Event, on_delete=models.CASCADE)
    name = models.CharField(max_length=100)
    email = models.EmailField()
    contact_number = models.CharField(max_length=15)
    registered_at = models.DateTimeField(auto_now_add=True)

    def __str__(self):
        return self.name

    if request.method == "POST":
        name = request.POST["name"]
